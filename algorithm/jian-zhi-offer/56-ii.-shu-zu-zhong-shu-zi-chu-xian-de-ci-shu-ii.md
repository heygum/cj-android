class Solution {
    public int singleNumber(int[] nums) {
        int m = 1, n = 0, out = 0;
        int max = 0, count = 1;
        for(int num:nums){
            if(max < num)
                max = num;
        }
        while(m <= max && m > 0){ 
            for(int num: nums){
            if((num & m) != 0)
                n++;
            }
            if((n % 3) == 1)
                out += m;
            n = 0;
            m <<= 1;
        }
        return out;
    }
}

class Solution {
    public int singleNumber(int[] nums) {
        int[] counts = new int[32];
        for(int num : nums) {
            for(int j = 0; j < 32; j++) {
                counts[j] += num & 1;
                num >>>= 1;
            }
        }
        int res = 0, m = 3;
        for(int i = 0; i < 32; i++) {
            res <<= 1;
            res |= counts[31 - i] % m;
        }
        return res;
    }
}
